∀(i) | 0<=i<=100 {
    ∀(j) | 0<=j<=100 {
        lu_u[i][j] = sigma(l[i][k] * u[k][j]) where 0<=k<=i-1
    }
}

∀(i) | 0<=i<=100 {
    ∀(j) | 0<=j<=100 {
        lu_l[i][j] = sigma(l[i][k] * u[k][j]) where 0<=k<=j-1
    }
}

∀(i) | 0<=i<=100 {
    ∀(j) | 0<=j<=100 {
        final_u[i][j] = a[i][j] - lu_u[i][j]
    }
}

∀(i) | 0<=i<=100 {
    ∀(j) | 0<=j<=100 {
        final_l[i][j] = (a[i][j] - lu_u[i][j])/final_u[j][j]
    }
}
