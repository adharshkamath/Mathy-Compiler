∀(k) | 0<=k<2600 {
    norm = Σ(A[i][k] * A[i][k]) | 0<=i<2000
    R[k][k] =  √(norm)
    ∀(i) | 0<=i<2000 {
        Q[i][k] = A[i][k] / R[k][k]
    }
    ∀(j) | k+1<=j<2600 {
        R[k][j] = Σ(Q[i][k] * A[i][j]) | 0<=i<2000
        ∀(j) | 0<=i<2000 {
            A[i][j] = A[i][j] - Q[i][k] * R[k][j]
        }
    }
}
