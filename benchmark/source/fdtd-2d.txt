        ∀(j) | 0<=j<100 {
            ey[i][j] = ey[i][j] - 0.5 * (hz[i][j] - hz[i-1][j])
        }
    }
    ∀(i) | 0<=i<150 {
        ∀(j) | 1<=j<100 {
            ex[i][j] = ex[i][j] - 0.5 * (hz[i][j] - hz[i][j-1])
        }
    }
    ∀(i) | 0<=i<149 {
        ∀(j) | 0<=j<99 {
            hz[i][j] = hz[i][j] - 0.7 *  (ex[i][j+1] - ex[i][j] + ey[i+1][j] - ey[i][j])
        }
    }
