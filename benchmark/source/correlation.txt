∀(x) | 0<=x<=100 {
    mean[x] = Σ(data[k][x]/100) | 0<=k<=100
}

∀(i) | 0<=i<=100 {
    ∀(j) | 0<=j<=100 {
        cov[i][j] = Σ(((data[k][i] - mean[i])*(data[k][j] - mean[j]))/99) | 0<=k<=100
    }
}

∀(t) | 0<=t<=100 {
    std_dev[t] = Σ(((data[k][t] - mean[t])*(data[k][t] - mean[t]))/99) | 0<=k<=100
}

∀(p) | 0<=p<=100 {
    std_dev[p] = √(std_dev[p])
}

∀(i) | 0<=i<=100 {
    ∀(j) | 0<=j<=100 {
        corr[i][j] = cov[i][j]/(std_dev[i]*std_dev[j])
    }
}
