alpha = 10
beta = 2

forall(i) where 0<=i<=50 {
    forall(j) where 0<=j<=50 {
        a_b[i][j] = sigma(alpha * (a[i][k] * b[k][j])) where 0<=k<=50
    }
}

forall(p) where 0<=p<=50 {
    forall(q) where 0<=q<=50 {
        c_b[p][q] = c[p][q]*beta
    }
}

forall(m) where 0<=m<=50 {
    forall(n) where 0<=n<=50 {
        c_out[m][n] = a_b[m][n] + c_b[m][n]
    }
}
