alpha = 10
beta = 2

∀(i) | 0<=i<=50 {
    ∀(j) | 0<=j<=50 {
        a_b[i][j] = Σ(alpha * (a[i][k] * b[k][j])) | 0<=k<=50
    }
}

∀(p) | 0<=p<=50 {
    ∀(q) | 0<=q<=50 {
        c_b[p][q] = c[p][q]*beta
    }
}

∀(m) | 0<=m<=50 {
    ∀(n) | 0<=n<=50 {
        c_out[m][n] = a_b[m][n] + c_b[m][n]
    }
}
